async function submitOrder(customerName: string, items: string, totalPrice: number) {
  try {
    const response = await fetch('http://your-server-address/api/place_order', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        customer_name: customerName,
        items: items,
        total_price: totalPrice
      }),
    });

    const data = await response.json();

    if (data.success) {
      alert(`คำสั่งถูกส่งเรียบร้อยแล้ว! หมายเลขคำสั่ง: ${data.order_id}`);
    } else {
      alert(`เกิดข้อผิดพลาด: ${data.error}`);
    }
  } catch (error) {
    console.error('Error:', error);
    alert('เกิดข้อผิดพลาดในการส่งคำสั่ง');
  }
}
